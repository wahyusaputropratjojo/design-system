## Installation

Install the following dependencies.

```bash tab="npm" title="Terminal"
npm install @base-ui/react tailwind-variants
```

```bash tab="pnpm" title="Terminal"
pnpm add @base-ui/react tailwind-variants
```

```bash tab="bun" title="Terminal"
bun add @base-ui/react tailwind-variants
```

Copy and paste the following code into `checkbox.tsx`.

<Files>
  <Folder name="components" defaultOpen disabled>
    <Folder name="base" defaultOpen disabled>
      <File name="checkbox.tsx" icon={<ReactIcon />} />
    </Folder>
  </Folder>
</Files>

```tsx lineNumbers title="checkbox.tsx"
import { Checkbox as CheckboxBase } from "@base-ui/react/checkbox";
import type { ComponentProps } from "react";
import { cn } from "@/lib/utils/tailwind-variants";

export type CheckboxProps = ComponentProps<typeof CheckboxBase.Root>;

export function Checkbox({ className, ...props }: CheckboxProps) {
  return (
    <CheckboxBase.Root
      className={cn(
        "flex size-5 cursor-pointer items-center justify-center rounded-md bg-neutral-3 focus-visible:outline-3 focus-visible:outline-primary-8 focus-visible:outline-offset-2 data-disabled:cursor-not-allowed data-readonly:cursor-default data-disabled:select-none data-checked:bg-primary-9 data-disabled:opacity-50",
        className,
      )}
      {...props}
    >
      <CheckboxBase.Indicator>
        <svg
          aria-hidden="true"
          className="size-3.5 shrink-0 stroke-2 text-white"
          fill="none"
          role="graphics-symbol"
          stroke="currentColor"
          strokeLinecap="round"
          strokeLinejoin="round"
          viewBox="0 0 24 24"
        >
          <path d="M20 6 9 17l-5-5" />
        </svg>
      </CheckboxBase.Indicator>
    </CheckboxBase.Root>
  );
}
```

## Usage

### Basic

A basic checkbox.

{
  <PreviewCard>
    <Checkbox />
  </PreviewCard>
}

```tsx title="Code"
<Checkbox />
```

### Default Checked

Use the `defaultChecked` prop to set the initial checked state in uncontrolled mode.

{
  <PreviewCard>
    <Checkbox defaultChecked />
  </PreviewCard>
}

```tsx title="Code"
// [!code word:defaultChecked]
<Checkbox defaultChecked />
```

### Disabled

Use the `disabled` prop to prevent user interaction and show the checkbox is not available.

{
  <PreviewCard>
    <div className="flex flex-col gap-2">
      <Checkbox disabled />
      <Checkbox disabled defaultChecked />
    </div>
  </PreviewCard>
}

```tsx title="Code"
// [!code word:disabled]
<Checkbox disabled />
<Checkbox disabled defaultChecked />
```

### Read Only

Use the `readOnly` prop to display the checkbox state without allowing changes.

{
  <PreviewCard>
    <div className="flex flex-col gap-2">
      <Checkbox readOnly />
      <Checkbox readOnly defaultChecked />
    </div>
  </PreviewCard>
}

```tsx title="Code"
// [!code word:readOnly]
<Checkbox readOnly />
<Checkbox readOnly defaultChecked />
```
