## Installation

Install the following dependencies.

```bash tab="npm" title="Terminal"
npm install tailwind-variants
```

```bash tab="pnpm" title="Terminal"
pnpm add tailwind-variants
```

```bash tab="bun" title="Terminal"
bun add tailwind-variants
```

Copy and paste the following code into `badge.tsx`.

<Files>
  <Folder name="components" defaultOpen disabled>
    <Folder name="base" defaultOpen disabled>
      <File name="badge.tsx" icon={<ReactIcon />} />
    </Folder>
  </Folder>
</Files>

```tsx lineNumbers title="badge.tsx"
import type { HTMLAttributes } from "react";
import type { VariantProps } from "tailwind-variants";
import { tv } from "@/lib/utils/tailwind-variants";

const badgeVariants = tv({
  base: "inline-flex items-center rounded-full border px-1.5 py-0.5 text-label-12",
  defaultVariants: {
    intent: "primary",
  },
  variants: {
    intent: {
      primary: "border-primary-7 bg-primary-3 text-neutral-12",
      secondary: "border-neutral-7 bg-neutral-3 text-neutral-12",
    },
  },
});

export type BadgeProps = {
  children: React.ReactNode;
} & HTMLAttributes<HTMLSpanElement> &
  VariantProps<typeof badgeVariants>;

export function Badge({ children, intent, ...props }: BadgeProps) {
  return (
    <span
      className={badgeVariants({
        intent,
      })}
      {...props}
    >
      {children}
    </span>
  );
}
```

## Usage

### Basic
A basic badge.

{
  <PreviewCard>
    <Badge>Badge</Badge>
  </PreviewCard>
}

```tsx title="Code"
<Badge>Badge</Badge>
```

### Intent
The `intent` prop defines the badge's visual style. Use `primary` for important information, and `secondary` for less prominent labels.

{
  <PreviewCard>
    <Badge intent="primary">Badge</Badge>
    <Badge intent="secondary">Badge</Badge>
  </PreviewCard>
}

```tsx title="Code"
// [!code word:intent]
<Badge intent="primary">Badge</Badge>
<Badge intent="secondary">Badge</Badge>
```
